SELECT I.PRODUCTID,I.COST
FROM INVENTORY I
WHERE I.COST>ALL(SELECT COST
FROM INVENTORY
WHERE PRODUCTID<>I.PRODUCTID);

SELECT C.NAME
FROM CUSTOMER C,STORE ST
WHERE C.STOREID=ST.STOREID
AND ST.STOREID IN(SELECT S.SUPPLIERID
FROM SUPPLIER S,MEDSUPPLIER M
WHERE M.SUPPLIERID=S.SUPPLIERID
AND S.NAME LIKE 'sup%'
AND M.STYPE='1'
AND M.ITEMS>20);

SELECT D1.NAME
FROM DOCTOR D1,PRESCRIBES P1
WHERE D1.LISCENSENO = P1.LISCENSENO AND
P1.CUSTOMERID IN (SELECT C1.CUSTOMERID FROM CUSTOMER C1
WHERE C1.BILLAMOUNT > ALL (SELECT BILLAMOUNT
FROM CUSTOMER
WHERE CUSTOMERID<>C1.CUSTOMERID));
